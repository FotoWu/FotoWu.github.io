Reorganize String

Description

Given a string S, check if the letters can be rearranged so that two characters that are adjacent to each other are not the same.

If possible, output any possible result. If not possible, return the empty string.

Example 1:

Input: S = "aab"
Output: "aba"

Example 2:

Input: S = "aaab"
Output: ""

Coding

/**
 * @param {string} S
 * @return {string}
 */
var reorganizeString = function(S) {
    if(S.length <= 1)
        return S;

    let output = "";
    let chars = {};

    for(var letter of S){
        chars[letter] = (letter in chars) ? chars[letter] + 1 : 1;
    }

    if(Math.max(...Object.values(chars)) > Math.ceil(S.length/2))
        return output;

    var entries = Object.entries(chars).sort((a, b) => (b[1] - a[1]));

    for(let entry of entries)
        output += entry[0].repeat(entry[1]);

    return spliceString(output);
};

var spliceString = function(str){
    var str1 = str.slice(0, Math.ceil(str.length/2));
    var str2 = str.slice(Math.ceil(str.length/2));

    var arr1 = str1.split("");
    var arr2 = str2.split("");

    let ptr1 = 1;
    for(let ptr2 = 0;ptr2<arr2.length;ptr2++){
        arr1.splice(ptr1, 0, arr2[ptr2]);
        ptr1 += 2;
    }

    return arr1.join("");
}
