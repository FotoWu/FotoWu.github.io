152. Maximum Product Subarray

Description

Given an integer array nums, find the contiguous subarray within an array (containing at least one number) which has the largest product.

Example 1:

Input: [2,3,-2,4]
Output: 6
Explanation: [2,3] has the largest product 6.

Example 2:

Input: [-2,0,-1]
Output: 0
Explanation: The result cannot be 2, because [-2,-1] is not a subarray.

Coding

  1. Brute Force: O(N2)
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxProduct = function(nums) {
    let maxProd = nums[0];
    let tmpProd = 1;

    for(let i=0;i<nums.length;i++){
        tmpProd = nums[i];
        for(let j=i+1;j<nums.length;j++){
            tmpProd *= nums[j];
            maxProd = Math.max(tmpProd, maxProd);
        }
        maxProd = Math.max(nums[i], maxProd);
    }

    return maxProd;
};
  1. Dynamic Programming: O(N)
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxProduct = function(nums) {
    let dp = new Array(nums.length);

    for(let i=0;i<nums.length;i++){
        if(i === 0){
            dp[i] = [nums[i], nums[i]];
        }
        else{
            let current = [
                Math.max(nums[i], nums[i]*dp[i-1][0], nums[i]*dp[i-1][1]),
                Math.min(nums[i], nums[i]*dp[i-1][0], nums[i]*dp[i-1][1])
            ];
            dp[i] = current;
        }
    }
    return Math.max(...dp.map(a => Math.max(a[0], a[1])));
};
