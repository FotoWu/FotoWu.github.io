32. Longest Valid Parentheses

Description

Given a string containing just the characters ‘(‘ and ‘)’, find the length of the longest valid (well-formed) parentheses substring.

Example 1:
Input: “(()”
Output: 2
Explanation: The longest valid parentheses substring is “()”

Example 2:
Input: “)()())”
Output: 4
Explanation: The longest valid parentheses substring is “()()”

Coding

  1. Stack
  • Time Complexity O(N)
  • Space Complexity O(N)
/**
 * @param {string} s
 * @return {number}
 */
var longestValidParentheses = function(s) {
    let count = 0;
    let stack = [-1];

    for(let i=0;i<s.length;i++){
        if(s[i]==='('){
            stack.push(i);
            continue;
        }

        stack.pop();
        if(!stack.length)
            stack.push(i);

        count = Math.max((i - stack[stack.length-1]), count);
    }

    return count;
};
  1. Two Pass: Using left to right pass then right to left pass
  • Time Complexity O(N)
  • Space Complexity O(1)
function longestValidParentheses(s){
  let longest = 0,leftCount = 0, rightCount = 0

  //left to right pass
  for(let i=0; i < s.length;i++){

    if(s[i] === '(')leftCount++
    else rightCount++

    if(leftCount === rightCount){
      longest = Math.max(longest,leftCount+rightCount)
    }else if(leftCount <rightCount){
      leftCount=0
      rightCount=0
    }
  }

  //right to left pass
  leftCount = 0, rightCount = 0
  for(let i=s.length-1; i >= 0;i--){

    if(s[i] === '(')leftCount++
    else rightCount++

    if(leftCount === rightCount){
      longest = Math.max(longest,leftCount+rightCount)
    }else if(leftCount >rightCount){
      leftCount=0
      rightCount=0
    }
  }
  return longest
}
